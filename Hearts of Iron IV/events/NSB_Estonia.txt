#Vapsid Civil War
country_event = {
	id = EST_events.7
	title = EST_events.7.t
	desc = EST_events.7.d
	picture = GFX_report_event_ger_civil_war
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			set_variable = {
				var = civil_war_size_var
				value = 0.9
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = party_popularity@fascism
			}
			random_owned_state = {
				limit = {
					has_state_flag = vaps_loyal_state
				}
				EST = { set_capital = { state = PREV } }
			}
			start_civil_war = {
				ruling_party = fascism
				ideology = neutrality
				size = civil_war_size_var
				states = all
				states_filter = {
					NOT = { has_state_flag = vaps_loyal_state }
				}
			}
			random_other_country = {
				limit = {
					original_tag = EST
					has_government = neutrality
				}
				save_event_target_as = neutral_EST_civil_war
			}
		}
	}
	
	option = { #Welp, that didn't go as planned...
		name = EST_events.7.a
		ai_chance = {
			factor = 10
		}
	}
}
