			}

				limit = {

				}

					size = 0.6
				}
			}

				limit = {
					has_idea = EST_vaps_idea_3
				}
				start_civil_war = {
					ideology = fascism
					size = 0.35
				}
			}
			else_if = {
				limit = {
					has_idea = EST_vaps_idea_2
				}
				add_popularity = {
					ideology = fascism
					popularity = -0.1
				}
				add_stability = -0.1
			}
			else = {
				add_popularity = {
					ideology = fascism
					popularity = -0.2
				}
			}
			remove_vaps_effect = yes
		}

		ai_will_do = {
			factor = 200
		}
	}
}
