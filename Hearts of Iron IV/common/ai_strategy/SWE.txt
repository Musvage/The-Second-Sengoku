SWE_war_with_ITA = {
	allowed = {
		original_tag = SWE
		has_dlc = "By Blood Alone"
	}
	enable = {
		check_variable = { ETH.host_nation = SWE }
		has_wargoal_against = ITA
	}
	abort = {
		NOT = { country_exists = ETH }
		NOT = { country_exists = ITA }
		NOT = { ETH = { is_government_in_exile = yes } }
	}

	ai_strategy = {
		type = antagonize
		id = "SWE"
		value = 200
	}
}
