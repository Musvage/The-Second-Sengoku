#scripted trigger for Yugoslavia

is_available_fighter_YUG = {
	OR = {
		AND = {
			has_tech = fighter1
			YUG = { 
				NOT = { has_tech = fighter1}
			}
		}
		AND = {
			has_tech = fighter2
			YUG = { 
				NOT = { has_tech = fighter2}
			}
		}
		AND = {
			has_tech = fighter3
			YUG = { 
				NOT = { has_tech = fighter3}
			}
		}
	}
	NOT = {
		has_war_with = YUG
	}
}
