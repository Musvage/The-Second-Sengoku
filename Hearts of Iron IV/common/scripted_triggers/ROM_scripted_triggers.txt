#scripted trigger for Hungary

is_available_fighter_ROM = {
	OR = {
		AND = {
			has_tech = fighter1
			ROM = { 
				NOT = { has_tech = fighter1}
			}
		}
		AND = {
			has_tech = fighter2
			ROM = { 
				NOT = { has_tech = fighter2}
			}
		}
		AND = {
			has_tech = fighter3
			ROM = { 
				NOT = { has_tech = fighter3}
			}
		}
	}
	NOT = {
		has_war_with = ROM
	}
}
