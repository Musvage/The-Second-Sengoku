AMO_raid_china_category = {
	AMO_raid_chinese_coast = {
	
		icon = generic_naval
	
		allowed = {
			tag = AMO
		}

		custom_cost_trigger = {
			has_manpower > 250
	   }
	   custom_cost_text = more_than_250_men_tt # localisation string

	   days_remove = 7 # Stays for 3 days before being removed

	   days_re_enable = 0 # Will show up in the interface and can be selected again after 100 days
		
		available = { 
			has_navy_size = {
				size > 0
			}

			NOT = {
				has_idea = AMO_bad_raid
				has_idea = AMO_decent_raid
				has_idea = AMO_great_raid
			}
		}
		
		visible = {

		}
		
		
		fixed_random_seed = no
		complete_effect = {
			hidden_effect = {
				add_manpower = -250
			}
		}

		remove_effect = {
			add_tech_bonus = {
				bonus = 0.05
				uses = 1
				category = naval_doctrine
			}
			navy_experience = 10
			hidden_effect = {
				random_list = {
					33 = {
						add_timed_idea = {
							idea = AMO_bad_raid
							days = 100
						}
					}
					33 = {
						add_timed_idea = {
							idea = AMO_decent_raid
							days = 100
						}
					}
					33 = {
						add_timed_idea = {
							idea = AMO_great_raid
							days = 100
						}
					}
					1 = {
						destroy_ships = {
							type = destroyer
							count = 1
						}
					}
				}
			}
			custom_effect_tooltip = AMO_raiding_tt
		}
