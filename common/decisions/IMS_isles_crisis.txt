IMS_isles_crisis_chi = {
	
	IMS_demand_compliance_chi = {

		icon = chi_trade_unions_demand

		allowed = {
			tag = CHI
		}

		available = {
			
			IMS = { has_government = neutrality }
		}

		targets = { IMS }

		target_trigger = {
			FROM = {
				tag = IMS
			}
		}

		days_remove = 20

		cost = 10

		fire_only_once = no

		ai_will_do = {
			factor = 0.03
			modifier = {
				
			}
			
		}

		complete_effect = {
		}

		remove_effect = {
			add_to_variable = { var = chi_isles_crisis_influence value = 5 }
			IMS = {
				set_variable = { var = chi_temp_trade_influence value = CHI.chi_isles_crisis_influence }
				subtract_from_variable = { var = chi_temp_trade_influence value = SHI.shi_isles_crisis_influence }
				set_variable = { var = shi_temp_trade_influence value = SHI.shi_isles_crisis_influence }
				subtract_from_variable = { var = shi_temp_trade_influence value = CHI.chi_isles_crisis_influence }
				add_to_variable = { var = isles_crisis_influence_pp_total value = 10 }
			}
			hidden_effect = {
				
			}
			custom_effect_tooltip = IMS_isles_crisis_decisions_tt
			effect_tooltip = { add_political_power = IMS.isles_crisis_influence_pp_total }
		}
	}
}
