world_stage_mongolia_category = {
	
	world_stage_mongolia_send_weapons = {

		icon = decision_generic_industry
		
		available = {
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		visible = {
			has_global_flag = world_stage_theater_mongolia_is_active
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		custom_cost_trigger = {
			has_equipment = { infantry_equipment > 999 } 
		}
		
		custom_cost_text = world_stage_1000_weapons_cost
		
		days_remove = 14
		
		complete_effect = {
			hidden_effect = {
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 1000
				}
			}
		}
		
		remove_effect  = {
			if = {
				limit = { tag = ROC }
				add_to_variable = {
					global.CHI_strength_mongolia_bonus = 5
					tooltip = world_stage_add_local_bonus
				}
			}
			if = {
				limit = { tag = FER }
				add_to_variable = {
					global.SOV_strength_mongolia_bonus = 5
					tooltip = world_stage_add_local_bonus
				}
			}
		}
	}
	
	world_stage_mongolia_send_forces = {

		icon = decision_generic_military
		
		available = {
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		visible = {
			has_global_flag = world_stage_theater_mongolia_is_active
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		custom_cost_trigger = {
			OR = {
				AND = {
					tag = FER
					check_variable = { global.world_stage_power_sov > 4 }
				}
				AND = {
					tag = ROC
					check_variable = { global.world_stage_power_chi > 4 }
				}
			}
		}
		
		custom_cost_text = world_stage_5_power_cost
		
		days_remove = 14
		
		remove_effect  = {
			if = {
				limit = { tag = ROC }
				add_to_variable = {global.world_stage_power_chi = -5 }
				add_to_variable = {
					global.CHI_strength_mongolia = 5
					tooltip = world_stage_add_local_power
				}
				
			}
			if = {
				limit = { tag = FER }
				add_to_variable = {global.world_stage_power_sov = -5 }
				add_to_variable = {
					global.SOV_strength_mongolia = 5
					tooltip = world_stage_add_local_power
				}
			}
		}
	}

	world_stage_mongolia_send_adviser = {

		icon = decision_generic_army_support
		
		available = {
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		visible = {
			has_global_flag = world_stage_theater_mongolia_is_active
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		custom_cost_trigger = {
			command_power > 24.99
		}
		
		custom_cost_text = world_stage_25_cp_cost
		
		days_remove = 14
		
		complete_effect = {
			hidden_effect = { add_command_power = -25 }
		}
		
		remove_effect  = {
			if = {
				limit = { tag = ROC }
				add_to_variable = {
					global.CHI_strength_mongolia_bonus = 5
					tooltip = world_stage_add_local_bonus
				}
			}
			if = {
				limit = { tag = FER }
				add_to_variable = {
					global.SOV_strength_mongolia_bonus = 5
					tooltip = world_stage_add_local_bonus
				}
			}
		}
	}
}
