world_stage_mongolia_category_first = {
	
	world_stage_mongolia_send_weapons_first = {

		icon = decision_generic_industry
		
		available = {
			has_global_flag = world_stage_theater_mongolia_is_active_first
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		visible = {
			has_global_flag = world_stage_theater_mongolia_is_active_first
			OR = {
				tag = FER
				tag = ROC
			}
		}
		
		custom_cost_trigger = {
			has_equipment = { infantry_equipment > 999 } 
		}
		
		custom_cost_text = world_stage_1000_weapons_cost
		
		days_remove = 14
		
		complete_effect = {
			hidden_effect = {
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = 1000
				}
			}
		}
		
		remove_effect  = {
			if = {
				limit = { tag = ROC }
				add_to_variable = {
					global.CHI_strength_mongolia_bonus = 5
					tooltip = world_stage_add_local_bonus
				}
			}
			if = {
				limit = { tag = FER }
				add_to_variable = {
					global.SOV_strength_mongolia_bonus = 5
					tooltip = world_stage_add_local_bonus
				}
			}
		}
	}

				add_to_variable = {
					global.SOV_strength_mongolia = 5
					tooltip = world_stage_add_local_power
