HIR_open_borders = {
	
	HIR_open_borders_decisions = {

		icon = chu_trade_unions_demand

		allowed = {
			tag = HIR
		}

		available = { 
			country_exists = HIR
			NOT = { has_country_flag = decision_active }
		}

		days_remove = 30

		cost = 50

		fire_only_once = no

		ai_will_do = {
			factor = 0.50
			modifier = {
				
			}
			
		}

		complete_effect = {
			set_country_flag = decision_active
		}

		remove_effect = {
			add_to_variable = { var = HIR.open_border_power value = 500 }
			add_to_variable = { var = HIR.open_border_army_power value = 100 }
			add_to_variable = { var = HIR.open_border_stability_power value = -0.010 }
			clr_country_flag = decision_active
		}
	}
	
	HIR_grant_citizenship_decisions = {

		icon = chu_trade_unions_demand

		allowed = {
			tag = HIR
		}

		available = { 
			country_exists = HIR
			check_variable = {
				HIR.open_border_power > 0
			}
			NOT = { has_country_flag = decision_active }
		}

		days_remove = 1

		cost = 0

		fire_only_once = no

		ai_will_do = {
			factor = 0.20
			modifier = {
				
			}
			
		}

		complete_effect = {
			set_country_flag = decision_active
		}

		remove_effect = {
			random_owned_controlled_state = {
				add_manpower = var:HIR.open_border_power
			}
			add_manpower = var:HIR.open_border_army_power
			add_stability = var:HIR.open_border_stability_power
			clear_variable = var:HIR.open_border_power
			clear_variable = HIR.open_border_stability_power
			clear_variable = var:HIR.open_border_army_power
			clr_country_flag = decision_active
		}
	}
}
