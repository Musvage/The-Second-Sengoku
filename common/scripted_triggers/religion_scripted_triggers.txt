has_religion_1 = {
	custom_trigger_tooltip = {
		tooltip = has_religion_buddhism_tt
		OR = {
			AND = {
				has_variable = country_religion
				check_variable = { country_religion = 1 }
			}
			AND = {
				has_variable = state_religion
				check_variable = { country_religion = 1 }
			}
		}
	}
}
