# THESE TRIGGERS ARE FOR ON-MAP TAGS
has_strength_alsace = {
	custom_trigger_tooltip = {
		tooltip = has_strength_here_tt
		OR = {
			AND = {
				check_variable = { world_stage_metropolis = 2 }
				check_variable = { global.ENG_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 3 }
				check_variable = { global.FRA_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 4 }
				check_variable = { global.SOV_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 5 }	has_country_flag = is_main_usa_colony
				check_variable = { global.USA_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 6 }
				check_variable = { global.NET_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 7 }
				check_variable = { global.PRT_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 8 }
				check_variable = { global.CHI_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 9 }
				check_variable = { global.GER_strength_alsace > 0 }
			}
			AND = {
				check_variable = { world_stage_metropolis = 10 }
				check_variable = { global.GER_strength_alsace > 0 }
			}
		}
		NOT = { check_variable = { world_stage_metropolis = 0} 
				check_variable = { world_stage_metropolis = 1} } 
	}
}
