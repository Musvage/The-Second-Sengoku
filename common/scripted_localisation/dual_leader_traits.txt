defined_text = {
	name = GetSecondaryTraitParagraph
	text = {
		trigger = {
			any_character = {
				is_country_leader = yes
				OR = {
					has_character_flag = coalition_with_fascism_secondary
					has_character_flag = untrustworthy_leader_secondary_desc
					has_character_flag = divine_authority_secondary
					has_character_flag = philantrophist_secondary
				}
			}
		}
		localization_key = diplo_paragraph
	}
	text = { localization_key = "" }
}
