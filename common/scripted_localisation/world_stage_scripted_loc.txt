defined_text = {
	name = GetTextColourAxis

	text = {	# is friendly to axis
		trigger = {
			OR = {
				check_variable = { world_stage_coalition = 3 }
				check_variable = { world_stage_coalition = 6 }
				AND = { tag = FER has_global_flag = world_stage_invasion_of_poland_molotov_ribbentrop }
			}
		}
		localization_key = use_green_text
	}
	text = {	# is neutral
		trigger = {
			NOT = { any_other_country = { check_variable = { world_stage_coalition = 3 } has_war_with = ROOT } }
			OR = {
				NOT = { has_variable = world_stage_coalition }
				check_variable = { world_stage_coalition = 100 }
				check_variable = { world_stage_coalition = 112 }
			}
		}
		localization_key = use_orange_text
	}
	text = {	# is axis' enemy, default
		localization_key = use_red_text
	}
}
