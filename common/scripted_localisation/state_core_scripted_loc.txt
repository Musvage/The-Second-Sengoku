defined_text = {
	name = Get_is_being_cored
	
	text = {
		trigger = { always = yes } # default!
		localization_key = state_is_not_being_cored
	}
	text = {
		trigger = {
			NOT = { is_core_of = CONTROLLER }
			state_and_controller_same_culture = yes
			is_fully_controlled_by = CONTROLLER
			compliance > 99.99
			resistance < 25
			OR = {
				has_neighbour_core = yes
				special_permission_for_islands = yes 	# code this later
			}
		}
		localization_key = state_is_being_cored
	}
}

defined_text = {
	name = Get_is_fully_controlled
	
	text = {
		trigger = { is_fully_controlled_by = CONTROLLER }
		localization_key = is_fully_controlled_positive
	}
	text = {
		trigger = { NOT = { is_fully_controlled_by = CONTROLLER } }
		localization_key = is_fully_controlled_negative
	}
}

defined_text = {
	name = Get_has_high_compliance
	
	text = {
		trigger = { compliance > 99.99 }

	}
	text = {
		trigger = { NOT = { compliance > 99.99 } }
		localization_key = has_high_compliance_negative
	}
}

defined_text = {
	name = Get_has_low_resistance
	
	text = {
		trigger = {	resistance < 25 }
		localization_key = has_low_resistance_positive
	}
	text = {
		trigger = { NOT = { resistance < 25 } }
		localization_key = has_low_resistance_negative
	}
}

defined_text = {
	name = Get_core_premission
	
	text = {
		trigger = {
			AND = {
				state_and_controller_same_culture = yes
				OR = {
					has_neighbour_core = yes
					special_permission_for_islands = yes
				}
			}
		}
		localization_key = core_premission_positive
	}
	text = {
		trigger = {
			NOT = {
				AND = {
					state_and_controller_same_culture = yes
					OR = {
						has_neighbour_core = yes
						special_permission_for_islands = yes
					}
				}
			}
		}
		localization_key = core_premission_negative
	}
}
