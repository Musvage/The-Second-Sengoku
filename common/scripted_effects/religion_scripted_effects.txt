# Scripted GUI Pie Chart example by Yard1
# Feel free to use in your mods, but give credits to Yard1 (both in code, with comments; and on your download page)
# this creates a pie chart of popularities - can be easily adapted to show anything that adds up to 100%
# modified by Maho
create_religion_pie_chart = {
	clear_array = religion_pie_chart
	clear_array = temp_pie

	for_each_loop = {
		array = religion_popularity_array

		add_to_array = { temp_pie = religion_popularity_array^i }
	}

	set_temp_variable = { temp = 0 }
	set_temp_variable = { temp_index = 1 }

	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_pie^num }
		}
		add_to_variable = { temp_pie^temp_index = temp_pie^temp }
		add_to_temp_variable = { temp = 1 }
		add_to_temp_variable = { temp_index = 1 }
	}

	set_temp_variable = { temp = 0 }

	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_pie^num }
		}
		resize_array = { array = religion_pie_chart value = religion_id_array^temp size = temp_pie^temp }
		add_to_temp_variable = { temp = 1 }
	}
}

religion_sort = {
	clear_array = temp_id_array

	for_each_loop = {
		array = religion_popularity_array

		set_temp_variable = { temp_id = religion_popularity_array^i }
		multiply_temp_variable = { temp_id = 1000 }
		add_to_temp_variable = { temp_id = religion_id_array^i }
		add_to_array = { temp_id_array = temp_id }
	}

	set_temp_variable = { array_sort = 1 }
	naive_sorting = yes

	set_temp_variable = { array_sort = 2 }
	naive_sorting = yes

	clear_array = religion_id_array

	for_each_loop = {
		array = temp_id_array

		modulo_variable = { temp_id_array^i = 1000 }
		add_to_array = { religion_id_array = temp_id_array^i }
	}
}

# Essentially a selection sort using find_lowest_in_array
# Implemented by Strategy Gamer
# I added a meta effect & scripted loc to take in array arguments - reimu
naive_sorting = {
	meta_effect = {
		text = {
			set_variable = { length_temp = [ARRAY]^num }
			for_loop_effect = {
				start = 0
				end = length_temp

				find_highest_in_array = {
					array = [ARRAY]
					value = max_value
					index = max_index
				}
				remove_from_array = {
					array = [ARRAY]
					index = max_index
				}
				add_to_array = { arr_temp = max_value }
			}
			clear_array = [ARRAY]
			for_each_loop = {
				array = arr_temp

				add_to_array = { [ARRAY] = v }
			}
			clear_array = arr_temp
			clear_variable = length_temp
			clear_variable = max_value
			clear_variable = max_index
		}
		ARRAY = "[ROOT.GetSortArray]"
	}
}
