# Scripted GUI Pie Chart example by Yard1
# Feel free to use in your mods, but give credits to Yard1 (both in code, with comments; and on your download page)
# this creates a pie chart of popularities - can be easily adapted to show anything that adds up to 100%
# modified by Maho
create_religion_pie_chart = {
	clear_array = religion_pie_chart
	clear_array = temp_pie
	set_temp_variable = { temp_total = 0 }

	for_each_loop = {
		array = religion_popularity_array

		add_to_array = { temp_pie = religion_popularity_array^i }
	}

	for_each_loop = {
		array = temp_pie

		round_variable = temp_pie^i
		add_to_temp_variable = { temp_total = temp_pie^i }
	}

	if = {
		limit = {
			check_variable = { temp_total > 100 }
		}
		subtract_from_temp_variable = { temp_total = 100 }
		find_highest_in_array = { array = temp_pie index = highest_index }
		subtract_from_variable = { temp_pie^highest_index = temp_total }
	}
	else_if = {
		limit = {
			check_variable = { temp_total < 100 }
		}
		set_temp_variable = { temp = 100 }
		subtract_from_temp_variable = { temp = temp_total }
		find_highest_in_array = { array = temp_pie index = highest_index }
		add_to_variable = { temp_pie^highest_index = temp }
	}

	set_temp_variable = { temp = 0 }
	set_temp_variable = { temp_index = 1 }

	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_pie^num }
		}
		add_to_variable = { temp_pie^temp_index = temp_pie^temp }
		add_to_temp_variable = { temp = 1 }
		add_to_temp_variable = { temp_index = 1 }
	}

	set_temp_variable = { temp = 0 }

	while_loop_effect = {
		limit = {
			check_variable = { temp < temp_pie^num }
		}
		resize_array = { array = religion_pie_chart value = religion_id_array^temp size = temp_pie^temp }
		add_to_temp_variable = { temp = 1 }
	}
}

add_religion_popularity = {
	log = "[?pop_value_added] added to [?pop_type_added] [THIS.GetName]"

	add_to_temp_variable = { pop_type_added = 1 }

	for_each_loop = {
		array = religion_id_array

		if = {
			limit = {
				check_variable = { religion_id_array^i = pop_type_added }
			}
			set_temp_variable = { pop_type_added = i }
			set_temp_variable = { break = 1 }
		}
	}

	set_temp_variable = { temp_hundred = religion_popularity_array^pop_type_added }
	add_to_temp_variable = { temp_hundred = pop_value_added }

	if = {
		limit = {
			check_variable = { temp_hundred > 100 }
		}
		subtract_from_temp_variable = { temp_hundred = 100 }
		subtract_from_temp_variable = { pop_value_added = temp_hundred }
	}

	for_loop_effect = {
		start = 0
		end = religion_popularity_array^num

		add_to_temp_array = { temp_percentage_popularity_array = v }
	}

	remove_from_temp_array = { array = temp_percentage_popularity_array value = pop_type_added }
	add_to_variable = { religion_popularity_array^pop_type_added = pop_value_added }
	#log = "[?religion_popularity_array^pop_type_added], [?pop_value_added] added to [THIS.GetName]"

	while_loop_effect = {
		limit = { check_variable = { pop_value_added > 0 } }

		for_loop_effect = {
			start = 0
			end = temp_percentage_popularity_array^num

			if = {
				limit = {
					check_variable = { religion_popularity_array^temp_percentage_popularity_array^v > 0 }
					check_variable = { pop_value_added > 0 }
				}
				subtract_from_variable = { religion_popularity_array^temp_percentage_popularity_array^v = 1 }
				subtract_from_temp_variable = { pop_value_added = 1 }
			}
		}
	}

	religion_sort = yes
	create_religion_pie_chart = yes
}

subtract_religion_popularity = {
	log = "[?pop_value_subtracted] subtracted from [?pop_type_subtracted] [THIS.GetName]"

	add_to_temp_variable = { pop_type_subtracted = 1 }

	for_each_loop = {
		array = religion_id_array

		if = {
			limit = {
				check_variable = { religion_id_array^i = pop_type_subtracted }
			}
			set_temp_variable = { pop_type_subtracted = i }
			set_temp_variable = { break = 1 }
		}
	}

	set_temp_variable = { temp_hundred = religion_popularity_array^pop_type_subtracted }
	subtract_from_temp_variable = { temp_hundred = pop_value_subtracted }

	if = {
		limit = {
			check_variable = { temp_hundred < 0 }
		}
		multiply_temp_variable = { temp_hundred = -1 }
		subtract_from_temp_variable = { pop_value_subtracted = temp_hundred }
	}

	for_loop_effect = {
		start = 0
		end = religion_popularity_array^num

		add_to_temp_array = { temp_percentage_popularity_array = v }
	}

	remove_from_temp_array = { array = temp_percentage_popularity_array value = pop_type_subtracted }
	subtract_from_variable = { religion_popularity_array^pop_type_subtracted = pop_value_subtracted }

	while_loop_effect = {
		limit = { check_variable = { pop_value_subtracted > 0 } }

		for_loop_effect = {
			start = 0
			end = temp_percentage_popularity_array^num

			if = {
				limit = {
					check_variable = { religion_popularity_array^temp_percentage_popularity_array^v < 100 }
					check_variable = { pop_value_subtracted > 0 }
				}
				add_to_variable = { religion_popularity_array^temp_percentage_popularity_array^v = 1 }
				subtract_from_temp_variable = { pop_value_subtracted = 1 }
			}
		}
	}

	religion_sort = yes
	create_religion_pie_chart = yes
}

religion_sort = {
	clear_array = temp_id_array

	for_each_loop = {
		array = religion_popularity_array

		set_temp_variable = { temp_id = religion_popularity_array^i }
		multiply_temp_variable = { temp_id = 1000 }
		add_to_temp_variable = { temp_id = religion_id_array^i }
		add_to_array = { temp_id_array = temp_id }
	}

	set_temp_variable = { array_sort = 1 }
	naive_sorting = yes

	set_temp_variable = { array_sort = 2 }
	naive_sorting = yes

	clear_array = religion_id_array

	for_each_loop = {
		array = temp_id_array

		modulo_variable = { temp_id_array^i = 1000 }
		add_to_array = { religion_id_array = temp_id_array^i }
	}
}

# Essentially a selection sort using find_lowest_in_array
# Implemented by Strategy Gamer
# I added a meta effect & scripted loc to take in array arguments - reimu
naive_sorting = {
	meta_effect = {
		text = {
			set_variable = { length_temp = [ARRAY]^num }
			for_loop_effect = {
				start = 0
				end = length_temp

				find_highest_in_array = {
					array = [ARRAY]
					value = max_value
					index = max_index
				}
				remove_from_array = {
					array = [ARRAY]
					index = max_index
				}
				add_to_array = { arr_temp = max_value }
			}
			clear_array = [ARRAY]
			for_each_loop = {
				array = arr_temp

				add_to_array = { [ARRAY] = v }
			}
			clear_array = arr_temp
			clear_variable = length_temp
			clear_variable = max_value
			clear_variable = max_index
		}
		ARRAY = "[ROOT.GetSortArray]"
	}
}

		}

update_country_pie_chart = {
	clear_array = religion_id_array
	clear_array = religion_popularity_array
	clear_variable = total_popularity

	resize_array = { array = religion_popularity_array value = 0 size = 5 }

	for_loop_effect = {
		start = 1
		end = 6

		add_to_array = { religion_id_array = v }
	}

	every_owned_state = {
		for_each_loop = {
			array = religion_id_array

			if = {
				limit = {
					check_variable = { religion_id_array^i = 1 }
				}
				set_temp_variable = { temp_id_1 = i }
			}
			else_if = {
				limit = {
					check_variable = { religion_id_array^i = 2 }
				}
				set_temp_variable = { temp_id_2 = i }
			}
			else_if = {
				limit = {
					check_variable = { religion_id_array^i = 3 }
				}
				set_temp_variable = { temp_id_3 = i }
			}
			else_if = {
				limit = {
					check_variable = { religion_id_array^i = 4 }
				}
				set_temp_variable = { temp_id_4 = i }
			}
			else_if = {
				limit = {
					check_variable = { religion_id_array^i = 5 }
				}
				set_temp_variable = { temp_id_5 = i }
			}
		}

		add_to_variable = { PREV.religion_popularity_array^0 = religion_popularity_array^temp_id_1 }
		add_to_variable = { PREV.religion_popularity_array^1 = religion_popularity_array^temp_id_2 }
		add_to_variable = { PREV.religion_popularity_array^2 = religion_popularity_array^temp_id_3 }
		add_to_variable = { PREV.religion_popularity_array^3 = religion_popularity_array^temp_id_4 }
		add_to_variable = { PREV.religion_popularity_array^4 = religion_popularity_array^temp_id_5 }
	}

	for_each_loop = {
		array = religion_id_array

		divide_variable = { religion_popularity_array^i = num_owned_states }
		round_variable = religion_popularity_array^i
		add_to_variable = { total_popularity = religion_popularity_array^i }
	}

	if = {
		limit = {
			check_variable = { total_popularity > 100 }
		}
		subtract_from_variable = { total_popularity = 100 }
		find_highest_in_array = { array = religion_popularity_array index = highest_index }
		subtract_from_variable = { religion_popularity_array^highest_index = total_popularity }
	}
	else_if = {
		limit = {
			check_variable = { total_popularity < 100 }
		}
		subtract_from_variable = { total_popularity = 100 }
		find_highest_in_array = { array = religion_popularity_array index = highest_index }
		subtract_from_variable = { religion_popularity_array^highest_index = total_popularity }
	}

	religion_sort = yes
	create_religion_pie_chart = yes
}
