EZO_civil_war_state_transfer_with_rebels = {
	if = {
		limit = {
			has_completed_focus = EZO_sellect_traditionalism
		}
		every_owned_state = {
			limit = {
				has_state_flag = EZO_player_revolt_flag
			}
			transfer_state_to = DZO
			add_core_of = DZO
			remove_core_of = EZO
		}
	}
	if = {
		limit = {
			has_completed_focus = EZO_sellect_strengthen_democratic_government
		}
		every_owned_state = {
			limit = {
				has_state_flag = EZO_player_revolt_flag
			}
			transfer_state_to = MZO
			add_core_of = MZO
			remove_core_of = EZO
		}
	}
}
