### THEATER RESOLUTIONS
resolve_mongolian_crisis = {
	# get losses: each side loses a proportion of its forces equal to: [opposite's side strength]/[both sides' strengths]. Example: a force with twice as strong as its enemy loses 1/3 of its total forces
	set_temp_variable = { total_strength = global.CHI_strength_mongolia_final }	# sum up both sides final strenghts
	add_to_temp_variable = { total_strength = global.SOV_strength_mongolia_final }
	set_temp_variable = { side_a_trickleback_proportion = global.CHI_strength_mongolia_final }	# temp variables do not need to be global
	divide_temp_variable = { side_a_trickleback_proportion = total_strength }
	set_temp_variable = { side_b_trickleback_proportion = global.SOV_strength_mongolia_final }
	divide_temp_variable = { side_b_trickleback_proportion = total_strength }
	set_variable = { global.CHI_strength_trickleback = global.CHI_strength_mongolia }	# tricklebacks equal to deployed strength multiplied by trickleback proportion
	multiply_variable = { global.CHI_strength_trickleback = side_a_trickleback_proportion }
	set_variable = { global.SOV_strength_trickleback = global.SOV_strength_mongolia }
	multiply_variable = { global.SOV_strength_trickleback = side_b_trickleback_proportion }
	add_to_variable = { global.world_stage_power_chi = global.CHI_strength_trickleback }
	add_to_variable = { global.world_stage_power_sov = global.SOV_strength_trickleback }
	log = "total_strength [?total_strength]"
	log = "CHI/SOV strentgh [?global.CHI_strength_mongolia]/[?global.SOV_strength_mongolia]"
	log = "CHI/SOV final strentgh [?global.CHI_strength_mongolia_final]/[?global.SOV_strength_mongolia_final]"
	log = "side_a/b_trickleback_proportion [?side_a_trickleback_proportion]/[?side_b_trickleback_proportion]"
	log = "CHI/SOV trickleback [?global.CHI_strength_trickleback]/[?global.SOV_strength_trickleback]"
	if = {
		limit = { check_variable = { global.CHI_strength_mongolia_final > global.SOV_strength_mongolia_final } }
		multiply_variable = { global.CHI_strength_mongolia_final = global.CHI_strength_mongolia_final }
	} else = { multiply_variable = { global.SOV_strength_mongolia_final = global.SOV_strength_mongolia_final } }
	random_list = {
		var:global.CHI_strength_mongolia_final = { country_event = worldstage.2 add_world_stage_power_chi_5 = yes }
		var:global.SOV_strength_mongolia_final = { country_event = worldstage.3 add_world_stage_power_sov_5 = yes }
	}
	clr_global_flag = world_stage_conflict_mongolia_on_going
	clr_global_variable = global.CHI_strength_mongolia_final
	clr_global_variable = global.CHI_strength_mongolia_final_final
	clr_global_variable = global.SOV_strength_mongolia_final
	clr_global_variable = global.SOV_strength_mongolia_final_final
}

world_stage_join_allies = {
	if = {
		limit = { OR = { tag = ROC tag = REI tag = PRT tag = OSA tag = NET tag = KOB tag = OKI tag = FER tag = AMC } }
		if = { limit = { tag = ROC } set_variable = { global.world_stage_coalition_chi = 1 } }
		if = { limit = { tag = REI } set_variable = { global.world_stage_coalition_ger = 1 } }
		if = { limit = { tag = PRT } set_variable = { global.world_stage_coalition_prt = 1 } }
		if = { limit = { tag = NET } set_variable = { global.world_stage_coalition_net = 1 } }
		if = { limit = { tag = OSA } set_variable = { global.world_stage_coalition_eng = 1 } }
		if = { limit = { tag = KOB } set_variable = { global.world_stage_coalition_fra = 1 } }
		if = { limit = { tag = OKI } set_variable = { global.world_stage_coalition_amc = 1 } }
		if = { limit = { tag = FER } set_variable = { global.world_stage_coalition_sov = 1 } }
		if = { limit = { tag = AMC } set_variable = { global.world_stage_coalition_amc = 1 } }
	} else = {
		set_variable = { world_stage_coalition = 1 }
	}
	world_stage_get_coalitions = yes
}
