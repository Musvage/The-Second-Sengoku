	log = "getting number of connections for [THIS.GetCurrentTheaterName] (held by state n.[?current_theater]"

		for_each_loop = {
			array = connects_to
			value = my_value
			index = my_index
			break = 4
			#effects
			set_variable = { ROOT.number_of_connections_displayed = my_index }
		}

}
