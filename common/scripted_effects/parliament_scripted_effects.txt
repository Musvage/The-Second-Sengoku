set_up_parliament = {
	clear_array = parliament_member_array

	set_variable = { parliament_seat_total = 465 }
	
	for_loop_effect = {
		start = 1
		end = 7

		add_to_array = { parliament_member_array = v }
	}

	parliament_semicircle = yes
	reload_parliament_gui = yes
}

reload_parliament_gui = {
	clear_array = parliament_main_array
	clear_array = parliament_seat_array

	add_to_array = { parliament_seat_array = party_popularity@democratic }
	add_to_array = { parliament_seat_array = party_popularity@democratic_socialism }
	add_to_array = { parliament_seat_array = party_popularity@communism }
	add_to_array = { parliament_seat_array = party_popularity@fascism }
	add_to_array = { parliament_seat_array = party_popularity@neutrality }
	add_to_array = { parliament_seat_array = party_popularity@autocracy }

	for_each_loop = {
		array = parliament_seat_array

		multiply_variable = { parliament_seat_array^i = parliament_seat_total }
		round_variable = parliament_seat_array^i

		add_to_temp_variable = { temp_total = parliament_seat_array^i }
	}

	if = {
		limit = {
			check_variable = { temp_total < parliament_seat_total }
		}
		set_temp_variable = { temp = parliament_seat_total }
		subtract_from_temp_variable = { temp = temp_total }
		find_highest_in_array = { array = parliament_seat_array index = highest_index }
		add_to_variable = { parliament_seat_array^highest_index = temp }
	}
	else_if = {
		limit = {
			check_variable = { temp_total > parliament_seat_total }
		}
		subtract_from_temp_variable = { temp_total = parliament_seat_total }
		find_highest_in_array = { array = parliament_seat_array index = highest_index }
		subtract_from_variable = { parliament_seat_array^highest_index = temp_total }
	}

	for_each_loop = {
		array = parliament_seat_array

		add_to_array = { temp_seat = parliament_seat_array^i }
	}

	set_temp_variable = { temp_index_1 = 0 }
	set_temp_variable = { temp_index_2 = 1 }

	set_temp_variable = { temp_size = temp_seat^num }
	subtract_from_temp_variable = { temp_size = 1 }

	while_loop_effect = {
		limit = {
			check_variable = { temp_index_1 < temp_size }
		}
		add_to_variable = { temp_seat^temp_index_2 = temp_seat^temp_index_1 }
		add_to_temp_variable = { temp_index_1 = 1 }
		add_to_temp_variable = { temp_index_2 = 1 }
	}

	for_loop_effect = {
		start = 0
		end = temp_seat^num

		resize_array = { array = parliament_main_array value = parliament_member_array^v size = temp_seat^v }
	}

	clear_array = temp_seat

	if = {
		limit = { has_variable = show_parliament_gui }
		add_to_variable = { show_parliament_gui = 1 }
	}
}
