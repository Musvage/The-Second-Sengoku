world_stage_get_get_theater_info_from_metropolis = {
	every_state = {
		limit = { has_variable = theater_owner }
		var:theater_owner = { set_variable = { PREV.theater_owner_coalition = world_stage_coalition } }
		log = "[THIS.GetTheaterName] (proxy: [THIS.GetName])"
		log = "theater_owner: [?theater_owner]"
		log = "world_stage_coalition: [?theater_owner_coalition]"
	}
} d_world_stage_get_get_theater_info_from_metropolis = { world_stage_get_get_theater_info_from_metropolis = yes log = "getting theater information from proxy metropolis to proxy theater" }

world_stage_get_get_tag_info_from_metropolis = {
	every_possible_country = {
		set_variable = { world_stage_metropolis = 100 }
		set_variable = { world_stage_coalition = 4 }
		set_variable = { world_stage_power = 0 }
		### TO-DO: ADD BELONGING TO FACTIONS LATER ### TO-DO: ADD BELONGING TO FACTIONS LATER ### TO-DO: ADD BELONGING TO FACTIONS LATER 
	}
	# starts with "hard coded" variables
	REI = { set_variable = { world_stage_metropolis = 1 } }	# for these ones it should be enough to get them once in start-up, the redundancy here works as safety mechanism
	OSA = { set_variable = { world_stage_metropolis = 2 } }
	KOB = { set_variable = { world_stage_metropolis = 3 } }
	FER = { set_variable = { world_stage_metropolis = 4 } }
	AMC = { set_variable = { world_stage_metropolis = 5 } }	#
	OKI = { set_variable = { world_stage_metropolis = 5 } }	# both are America	(NOTE: possibility of OKI becoming independent and not being USA anymore should be considered here)
	ROC = { set_variable = { world_stage_metropolis = 6 } }
	HIR = { set_variable = { world_stage_metropolis = 7 } }
	PRT = { set_variable = { world_stage_metropolis = 8 } }
	#KIT = { set_variable = { world_stage_metropolis = 9 } }	# Italy, not implemented
	#SPR = { set_variable = { world_stage_metropolis = 11 } }	# Spain, not implemented
	random_country = {
		limit = { has_idea = all_internationally_recognised }
		set_variable = { world_stage_metropolis = 10 }
	}
	# now imports variables from metropolis
	every_country = {
		limit = { check_variable = { world_stage_metropolis > 0 } }
		var:world_stage_metropolis = {
			set_variable = { PREV.world_stage_power = world_stage_power }
			set_variable = { PREV.world_stage_coalition = world_stage_coalition }
		}
		log = "world_stage_metropolis: [?world_stage_metropolis]"
		log = "world_stage_coalition: [?world_stage_coalition]"
		log = "world_stage_power: [?world_stage_power]"
	}
} d_world_stage_get_get_tag_info_from_metropolis = { world_stage_get_get_tag_info_from_metropolis = yes log = "get all tag info from metropolis" }

				}
			}
		}
