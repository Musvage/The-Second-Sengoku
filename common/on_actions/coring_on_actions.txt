on_actions = {
	
	on_state_control_changed = {
		effect = {
			set_variable = { coring_progress = 0 }
		}
	}
	
	on_weekly = {
		effect = {
			every_controlled_state = {
				if = {
					limit = {
						check_variable = { coring_progress > 99.99
						NOT = { is_core_of = ROOT }
					}
					add_core_of = ROOT
				}
				if = {
					limit = {
						NOT = { is_core_of = ROOT }
						# triggers for coring
						state_and_controller_same_culture = yes
						OR = {
							# is neighbour, code this later
							special_permission_for_islands = yes 	# code this later
						}
					}
					add_to_variable = { coring_progress = 10 }	# arbitrary number, can change later
				}
			}
		}
	}
}
