on_actions = {
	on_government_change = {
		effect = {
			log = "[THIS.GetName] on_government_change "
			if = {
				limit = { OR = { has_government = democratic has_government = democratic_socialism } }
				country_event = flavor_generic.1
			}
			if = {
				limit = { has_government = communism }
				country_event = flavor_generic.2
			}
			if = {
				limit = { OR = { has_government = fascism has_government = autocracy has_government = neutrality } }
				country_event = flavor_generic.3
			}
		}
	}
	on_send_volunteers = {	# ROOT is sender, FROM is receiver.
		effect = {
			ROOT = { FROM = { country_event = flavor_generic.4 } }
		}
	}
}
