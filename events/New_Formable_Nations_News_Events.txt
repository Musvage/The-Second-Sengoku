# Formation of Slavic Federation
news_event = {
	id = wtt_news.1500
	title = wtt_news.1500.t
	desc = wtt_news.1500.desc
	picture = GFX_news_event_generic_parliament

	is_triggered_only = yes
	
	option = {
		name = wtt_news.1500.a
		trigger = {
			OR = {
				original_tag = POL

		name = wtt_news.1500.b
