country_event = { # Spawn Spain
    id = spain.2
	title = spain.2.t
	desc = spain.2.d
	picture = GFX_report_event_generic_sign_treaty2 #FIXME

	fire_only_once = yes
    trigger = {
        AND = {
            date > 1939.4.1
            original_tag = FER
            date < 1940.1.1
            SPR_enrique_lister = {
                is_leading_army = yes
            }

            TOH = {
                has_opinion = {
                    target = FER
                    value > 50
                }
            }
        }
    }

	option = {
		ai_will_do = { factor = 0 }
		name="May they rise again (Play as Spain)"

        TOH = {
            country_event = spain.3
        }
	}

    option = {
        ai_will_do = { factor = 1000 }
        name="Stop this folly at once"
    }
}
