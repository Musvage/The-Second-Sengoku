# ROC completes 'ROC_aid_mongolia'
news_event = {
	id = worldstage.1
	title = worldstage.1.t
	desc = worldstage.1.d 
	picture = GFX_news_great_depression_over

	is_triggered_only = yes

	major = yes
	
	option = {
		name = worldstage.1.axis
		trigger = {
			has_government = fascism
		}
		if = {
			limit = { OR = { tag = FER tag = ROC } }
			effect_tooltip = { activate_mission = world_stage_mongolia_main_mission }
		}
	}
	option = {
		name = worldstage.1.komin
		trigger = {
			has_government = communism
		}
		if = {
			limit = { OR = { tag = FER tag = ROC } }
			effect_tooltip = { activate_mission = world_stage_mongolia_main_mission }
		}
	}
	option = {
		name = worldstage.1.other
		trigger = {
			NOT = {
				has_government = fascism
				has_government = communism
			}
		}
	}
}

# ROC completes 'ROC_aid_mongolia', hidden event (looks convoluted, but this is to guarantee the final event will fire if both FER and ROC no longer exist fater 90 days)
news_event = {
	id = worldstage.2
	title = "worldstage.2"
	desc = "this event shuold not appear!"
	picture = GFX_news_great_depression_over

	is_triggered_only = yes

	hidden = yes
	
	immediate = {
		if = {
			limit = {
				NOT = { has_global_flag = world_stage_flag_mongolia }
			}
			if = {
				limit = { check_variable = { global.ROC_strength_mongolia > global.SOV_strength_mongolia } }
				random_country = { country_event =  worldstage.3 }
				ROC = {
					add_named_threat = { 
						threat = 0.25 
						name = ROC_threat_aid_mongolia_final
					}
				}
			} else = {
				random_country = { country_event =  worldstage.4 }
				FER = {
					add_named_threat = { 
						threat = 0.25 
						name = ROC_threat_aid_mongolia_final
					}
				}
			}
			set_global_flag = world_stage_flag_mongolia
		}
	}
}
