#Chugoku enemy or ally
country_event = {
	id = chugoku.8
	title = chugoku.8.t
	desc = chugoku.8.desc
	picture = GFX_report_event_generic_sign_treaty2

	is_triggered_only = yes
	
	option = {
		name = chugoku.8.a #Chugoku has become enemy!
		add_opinion_modifier = {
			target = CHU
			modifier = CHU_new_enemy
		}
		CHU = {
			add_opinion_modifier = {
				target = ROOT
				modifier = CHU_new_enemy
			}
		}
		hidden_effect = {
			CHU = { country_event = { id = chugoku.9 hours = 6 } }
			add_ideas = CHU_has_enemy
		}
	}
	option = {
		name = chugoku.8.b #Chugoku has become friend!
		diplomatic_relation = {
			country = CHU
			relation = non_aggression_pact
			active = yes
		}
		add_opinion_modifier = {
			target = CHU
			modifier = CHU_ally
		}
		CHU = {
			add_opinion_modifier = {
				target = ROOT
				modifier = CHU_ally
			}
		}
		hidden_effect = {
			CHU = { country_event = { id = chugoku.10 hours = 6 } }
			add_ideas = CHU_has_friend
		}
	}
}
