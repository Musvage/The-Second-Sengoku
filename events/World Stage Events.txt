# ROC wins in Mongolia
news_event = {
	id = worldstage.2
	title = worldstage.2.t
	desc = worldstage.2.d
	picture = GFX_news_great_depression_over

	is_triggered_only = yes

	major = yes
	
	immediate = {
		if = {
			limit = { has_global_flag = league_of_nations_exists NOT = { has_global_flag = china_aligned_axis } }
			set_variable = { global.theater_owner_mongolia = 5 }
		}
		if = {
			limit = { has_global_flag = league_of_nations_exists has_global_flag = china_aligned_axis }
			set_variable = { global.theater_owner_mongolia = 6 }
		}
		if = {
			limit = { NOT = { has_global_flag = league_of_nations_exists has_global_flag = china_aligned_axis } }
			set_variable = { global.theater_owner_mongolia = 4 }
		}
		if = {
			limit = { NOT = { has_global_flag = league_of_nations_exists } has_global_flag = china_aligned_axis }
			set_variable = { global.theater_owner_mongolia = 3 }
		}
	}
	
	option = {
		name = worldstage.2.axis
		trigger = {
			OR = { has_government = fascism tag = ROC }
		}
		effect_tooltip = {
			add_world_stage_power_chi_5 = yes
		}
	}
	option = {
		name = worldstage.2.komin
		trigger = {
			OR = { has_government = communism tag = FER }
		}
		effect_tooltip = {
			add_world_stage_power_chi_5 = yes
		}
	}
	option = {
		name = worldstage.2.other
		trigger = {
			NOT = {
				has_government = communism
				has_government = fascism
				tag = ROC
				tag = FER
			}
		}
		effect_tooltip = {

		}
	}
}

### INVASION OF POLAND

# Invasion of Poland - Start
country_event = {
	id = worldstage.15
	title = worldstage.15.t
	desc = worldstage.15.d 
	picture = GFX_news_great_depression_over

	is_triggered_only = yes

	major = yes
	
	immediate = {
		if = {
			limit = {
				country_exists = OSA
				country_exists = KOB
				NOT = { KOB = { is_in_faction_with = OSA } }
			}
			OSA = { add_to_faction = KOB }
		}
		every_country = {
			if = {
				limit = { check_variable = { world_stage_coalition = 6 } }
				set_variable = { world_stage_coalition = 3 }				# Pact of Steel becomes Axis
			}
			if = {
				limit = { OR = { original_tag = OSA original_tag = KOB } }
				set_variable = { world_stage_coalition = 2 }				# LoN becomes Allies
			}
		}
	}
	
	option = {
		name = worldstage.15.axis
		trigger = { has_world_stage_coalition_axis = yes }
	}
	option = {
		name = worldstage.15.komin1
		trigger = {
			has_world_stage_coalition_komintern = yes
			NOT = { has_global_flag = GER_SOV_pact_enabled }
		}
	}
	option = {
		name = worldstage.15.komin2
		trigger = {
			has_world_stage_coalition_komintern = yes
			has_global_flag = GER_SOV_pact_enabled
		}
	}
	option = {
		name = worldstage.15.allies
		trigger = { has_world_stage_coalition_allies = yes }

# The End of the League of Nations
news_event = {
	id = worldstage.55
	title = worldstage.55.t
	desc = worldstage.55.d 
	picture = GFX_news_great_depression_over

	is_triggered_only = yes

	major = yes
	
	immediate = {
		log = "Begone!, League of Nations!"
		every_country = {
			limit = { has_idea = SS_lon_member }
			remove_ideas = SS_lon_member
		}
		every_country = {
			limit = { has_idea = SS_lon_member_japan }
			remove_ideas = SS_lon_member_japan
		}
		clr_global_flag = league_of_nations_exists
		set_global_flag = SOV_already_expelled
		# switch countries' local factions							# (international factions must always be set to ww2 parameters and are used only in the background)
		OSA = { set_variable = { world_stage_coalition = 1 } }
		KOB = { set_variable = { world_stage_coalition = 1 } }
		FER = { set_variable = { world_stage_coalition = 2 } }		# URSS was already expelled from LoN at this point but we'll run the command again just in case the FER player did not play his part 
		REI = { set_variable = { world_stage_coalition = 3 } }
		if = {
			limit = { has_global_flag = netherlands_aligned_allies }
			set_variable = { global.world_stage_coalition_net = 1 }
			HIR = { set_variable = { world_stage_coalition = 1 } }
		}
		if = {
			limit = { has_global_flag = netherlands_aligned_axis }
			set_variable = { global.world_stage_coalition_net = 3 }
			HIR = { set_variable = { world_stage_coalition = 3 } }
		}
		if = {
			limit = { NOT = { has_global_flag = netherlands_aligned_axis has_global_flag = netherlands_aligned_allies } }
			set_variable = { global.world_stage_coalition_net = 0 }
			HIR = { set_variable = { world_stage_coalition = 0 } }
		}
		if = {
			limit = { has_global_flag = portugal_aligned_allies }
			set_variable = { global.world_stage_coalition_prt = 1 }
			PRT = { set_variable = { world_stage_coalition = 1 } }
		}
		if = {
			limit = { has_global_flag = portugal_aligned_axis }
			set_variable = { global.world_stage_coalition_prt = 3 }
			PRT = { set_variable = { world_stage_coalition = 3 } }
		}
		if = {
			limit = { NOT = { has_global_flag = portugal_aligned_axis has_global_flag = portugal_aligned_allies } }
			set_variable = { global.world_stage_coalition_prt = 0 }
			PRT = { set_variable = { world_stage_coalition = 0 } }
		}
		if = {
			limit = { has_global_flag = china_aligned_allies }
			set_variable = { global.world_stage_coalition_chi = 1 }
			ROC = { set_variable = { world_stage_coalition = 1 } }
		}
		if = {
			limit = { has_global_flag = china_aligned_komintern }
			set_variable = { global.world_stage_coalition_chi = 2 }
			ROC = { set_variable = { world_stage_coalition = 2 } }
		}
		if = {
			limit = { has_global_flag = china_aligned_axis }
			set_variable = { global.world_stage_coalition_chi = 3 }
			ROC = { set_variable = { world_stage_coalition = 3 } }
		}
		if = {
			limit = { NOT = { has_global_flag = china_aligned_axis has_global_flag = china_aligned_allies has_global_flag = china_aligned_komintern } }
			set_variable = { global.world_stage_coalition_chi = 0 }
			ROC = { set_variable = { world_stage_coalition = 0 } }
		}
		every_country = {
			limit = { has_world_stage_coalition_allies = yes }
			set_variable = { world_stage_coalition = 1 }
		}
		every_country = {
			limit = { has_world_stage_coalition_komintern = yes }
			set_variable = { world_stage_coalition = 2 }
		}
		every_country = {
			limit = { has_world_stage_coalition_axis = yes }
			set_variable = { world_stage_coalition = 3 }
		}
		every_country = {
			limit = { has_world_stage_coalition_neutrality = yes }
			set_variable = { world_stage_coalition = 4 }
		}
		# create on-map factions
		if = {
			limit = { country_exists = REI is_in_faction = no }
			REI = {
				create_faction = "Axis"
				set_variable = { world_stage_coalition = 3 }
				every_other_country = {
					limit = { has_world_stage_coalition_axis = yes is_in_faction = no }
					REI = { add_to_faction = PREV }
					set_variable = { world_stage_coalition = 3 }
				}
			}
		} else = {
			if = {
				limit = { country_exists = ROC is_in_faction = no has_world_stage_coalition_axis = yes }
				ROC = {
					create_faction = "Axis"
					set_variable = { world_stage_coalition = 3 }
					every_other_country = {
						limit = { has_world_stage_coalition_axis = yes is_in_faction = no }

						set_variable = { world_stage_coalition = 3 }
					}
				}

				random_country = {
					limit = { has_world_stage_coalition_axis = yes is_in_faction = no }
					create_faction = "Axis"
					set_variable = { world_stage_coalition = 3 }
					set_temp_variable = { leader_id = THIS.id }
					every_other_country = {
						limit = { has_world_stage_coalition_axis = yes }
						set_temp_variable = { other_id = THIS.id }

					set_variable = { world_stage_coalition = 2 }
				}
			}
		}
		if = {
			limit = { country_exists = OSA is_in_faction = no }
			OSA = {
				log = "OSA creates Allies"
				create_faction = "Allies"
				set_variable = { world_stage_coalition = 1 }
				every_other_country = {
					limit = { has_world_stage_coalition_allies = yes is_in_faction = no }

	title = worldstage_philipines.1.t
	desc = worldstage_philipines.1.d 
	picture = GFX_news_great_depression_over

	is_triggered_only = yes

	major = yes
	
	immediate = {														# NEXT THEATER: england
		#set_variable = { global.theater_owner_philipines = 1 }	# philipines is always "controlled" by allies
		set_global_flag = world_stage_theater_england_is_active
		clr_global_flag = world_stage_theater_philipines_is_active
		add_to_variable = { global.world_stage_power_ger = global.GER_strength_philipines }
		add_to_variable = { global.world_stage_power_fra = global.FRA_strength_philipines }
		add_to_variable = { global.world_stage_power_sov = global.SOV_strength_philipines }
		add_to_variable = { global.world_stage_power_eng = global.ENG_strength_philipines }
		add_to_variable = { global.world_stage_power_usa = global.USA_strength_philipines }
		add_to_variable = { global.world_stage_power_net = global.NET_strength_philipines }
		add_to_variable = { global.world_stage_power_prt = global.PRT_strength_philipines }
		add_to_variable = { global.world_stage_power_chi = global.CHI_strength_philipines }
		add_to_variable = { global.tc_base_chiba_rubber = -1 }
		#add_to_variable = { global.tc_base_okinawa_rubber = -1 }
		add_to_variable = { global.tc_base_fengtian_rubber = 1 }
		every_country = {  check_trade_cities = yes }
	}
	
	option = {
		name = worldstage_philipines.1.axis
		trigger = { has_world_stage_coalition_axis = yes }
		custom_effect_tooltip = gain_philipines_axis
		custom_effect_tooltip = lose_philipines_ally
	}
	option = {
		name = worldstage_philipines.1.komin
		trigger = { has_world_stage_coalition_komintern = yes }
		custom_effect_tooltip = gain_philipines_axis
		custom_effect_tooltip = lose_philipines_ally
	}
	option = {
		name = worldstage_philipines.1.allies
		trigger = { has_world_stage_coalition_allies = yes }
		custom_effect_tooltip = gain_philipines_axis
		custom_effect_tooltip = lose_philipines_ally
	}
	option = {
		name = worldstage_philipines.1.other
		trigger = { has_world_stage_coalition_unaligned = yes }
		custom_effect_tooltip = gain_philipines_axis
		custom_effect_tooltip = lose_philipines_ally
	}
}
