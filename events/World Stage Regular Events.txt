country_event = {															# AXIS WINS					
	id = worldstage_alsace.1
	title = worldstage_alsace.1.t
	desc = worldstage_alsace.1.d 
	picture = GFX_news_great_depression_over

	is_triggered_only = yes

	major = yes show_major = { NOT = { has_country_flag = hide_worldstage_notification } }
	
	immediate = {														# NEXT THEATER: france and/or alps
		set_variable = { global.theater_owner_alsace = 3 }
		clr_global_flag = world_stage_theater_alsace_is_active
		set_temp_variable = { chi_forces_value = global.CHI_strength_alsace }
		set_temp_variable = { ger_forces_value = global.GER_strength_alsace }
		set_temp_variable = { usa_forces_value = global.USA_strength_alsace }
		set_temp_variable = { sov_forces_value = global.SOV_strength_alsace }
		set_temp_variable = { fra_forces_value = global.FRA_strength_alsace }
		set_temp_variable = { eng_forces_value = global.ENG_strength_alsace }
		set_temp_variable = { net_forces_value = global.NET_strength_alsace }
		set_temp_variable = { prt_forces_value = global.PRT_strength_alsace }
		# new defender forces
		set_temp_variable = { forces_value = global.world_stage_power_fra }
		divide_temp_variable = { forces_value = 5 }
		clamp_temp_variable = { var = forces_value min = 1 max = 25 }
		if = {
			limit = {
				NOT = { check_variable = { global.theater_owner_france = 3 }
						check_variable = { global.theater_owner_alpes = 3 } }	# both are not controlled by axis
			}
			divide_temp_variable = { chi_forces_value = 2 }
			divide_temp_variable = { prt_forces_value = 2 }
			divide_temp_variable = { net_forces_value = 2 }
			divide_temp_variable = { usa_forces_value = 2 }
			divide_temp_variable = { fra_forces_value = 2 }
			divide_temp_variable = { sov_forces_value = 2 }
			divide_temp_variable = { ger_forces_value = 2 }
			divide_temp_variable = { eng_forces_value = 2 }
		}
		if = {
			limit = { NOT = { check_variable = { global.theater_owner_france = 3 } } }
			set_global_flag = world_stage_theater_france_is_active
			add_to_variable = { global.GER_strength_france = ger_forces_value }
			add_to_variable = { global.FRA_strength_france = fra_forces_value }
			add_to_variable = { global.SOV_strength_france = sov_forces_value }
			add_to_variable = { global.ENG_strength_france = eng_forces_value }
			add_to_variable = { global.USA_strength_france = usa_forces_value }
			add_to_variable = { global.NET_strength_france = net_forces_value }
			add_to_variable = { global.PRT_strength_france = prt_forces_value }
			add_to_variable = { global.CHI_strength_france = chi_forces_value }
			subtract_from_variable = { global.world_stage_power_fra = forces_value }
			add_to_variable = { global.FRA_strength_france = forces_value }
		}
		if = {
			limit = { NOT = { check_variable = { global.theater_owner_alpes = 3 } } }
			set_global_flag = world_stage_theater_alpes_is_active
			add_to_variable = { global.GER_strength_alpes = ger_forces_value }
			add_to_variable = { global.FRA_strength_alpes = fra_forces_value }
			add_to_variable = { global.SOV_strength_alpes = sov_forces_value }
			add_to_variable = { global.ENG_strength_alpes = eng_forces_value }
			add_to_variable = { global.USA_strength_alpes = usa_forces_value }
			add_to_variable = { global.NET_strength_alpes = net_forces_value }
			add_to_variable = { global.PRT_strength_alpes = prt_forces_value }
			add_to_variable = { global.CHI_strength_alpes = chi_forces_value }
			subtract_from_variable = { global.world_stage_power_fra = forces_value }
			add_to_variable = { global.FRA_strength_france = forces_value }
		}
		set_variable = { global.GER_strength_alsace = 0 }
		set_variable = { global.FRA_strength_alsace = 0 }
		set_variable = { global.SOV_strength_alsace = 0 }
		set_variable = { global.ENG_strength_alsace = 0 }
		set_variable = { global.USA_strength_alsace = 0 }
		set_variable = { global.NET_strength_alsace = 0 }
		set_variable = { global.PRT_strength_alsace = 0 }
		set_variable = { global.CHI_strength_alsace = 0 }
		if = {
			limit = { check_variable = { global.AXIS_locals_alsace  < 1 } }
			set_variable = { global.AXIS_locals_alsace = 1 }
		}
	}
	
	option = {
		name = worldstage_alsace.1.axis
		trigger = { has_world_stage_coalition_axis = yes }
		effect_tooltip = {
			
		}
	}
	option = {

		effect_tooltip = {
