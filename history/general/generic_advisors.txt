every_possible_country = {	### COMMUNIST ADVISER
	limit = {
		NOT = { original_tag = REI original_tag = OSA original_tag = KOB original_tag = HIR original_tag = AMC original_tag = OKI original_tag = PRT original_tag = FER original_tag = SPR original_tag = KIT original_tag = WIT }
		NOT = { original_tag = IMS }
		is_dynamic_country = no
		NOT = {
			has_allowed_idea_with_traits = { idea = communist_revolutionary characters = yes }
			has_allowed_idea_with_traits = { idea = marxist_fundamentalist characters = yes }
			has_allowed_idea_with_traits = { idea = colonial_communist characters = yes }
		}
	}
	generate_character = {
		token_base = generic_communist_revolutionary_japan # token will be TAG_token_base
		advisor = {
			slot = political_advisor
			traits = { communist_revolutionary }
			visible = { OR = { has_government = communism has_country_flag = allowed_path_communism } }
			ai_will_do = {
				factor = 1
				modifier = {
					factor = 0
					has_country_flag = initial_ideology_communism
					NOT = { has_country_flag = allowed_path_communism }
				}
			}
		}
		portraits = { army = { small = "GFX_Portrait_Japan_Communist_Revolutionary_Small" } }
	}
}
